var myForm;
var t = {
    "labelWidth": false,
    "labelAlign": false,
    "inputWidth": false
};
function initForm() {
    var formData = [{
        type: "settings",
        position: "label-left",
        labelWidth: 100,
        inputWidth: 120
    }, {
        type: "input",
        label: "Login",
        value: "login"
    }, {
        type: "password",
        label: "Password",
        value: "123"
    }, {
        type: "checkbox",
        label: "Remember me",
        checked: true
    }];
    for (var a in t) {
        if (!t[a])
            t[a] = document.getElementById(a);
        formData[0][a] = t[a].options[t[a].selectedIndex].value;
        if (!isNaN(parseInt(formData[0][a])))
            formData[0][a] = parseInt(formData[0][a]);
    }
    if (myForm)
        myForm.unload();
    myForm = new dhtmlXForm("myForm", formData);
}
</script>
