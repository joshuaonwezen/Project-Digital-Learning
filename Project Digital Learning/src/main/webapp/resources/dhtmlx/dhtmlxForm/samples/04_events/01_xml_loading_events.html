var myForm,
logObj;
function doOnLoad() {
    myForm = new dhtmlXForm("myForm");
    myForm.attachEvent("onXLS", function() {
        logEvent("onXLS event called<br>");
    });
    myForm.attachEvent("onXLE", function() {
        logEvent("onXLE event called<br>");
    });
    myForm.loadStruct("../common/dhxform.xml?e=" + new Date().getTime());
}
function logEvent(t) {
    if (!logObj)
        logObj = document.getElementById("simpleLog");
    logObj.innerHTML += t;
}
function clearLog() {
    if (!logObj)
        logObj = document.getElementById("simpleLog");
    logObj.innerHTML = "";
}
</script>
