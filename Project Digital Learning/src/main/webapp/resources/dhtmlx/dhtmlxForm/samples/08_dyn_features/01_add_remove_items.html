        }, {
        type: "button",
        name: "button",

    var itemData = {

            type: "select",
            name: "video_codec",

            validate: "NotEmpty",
            options: [{
                text: "DivX",
                value: "DivX"
            }, {
                text: "XviD",
                value: "XviD",
                selected: true
            }]
            }, {
            type: "select",
            name: "video_bitrate",
            label: "Bitrate",
            validate: "ValidTime",
            options: [{
                text: "728 kbps",
                value: "728"
            }, {
                text: "1226 kbps",
                value: "1226",
                selected: true
            }, {
                text: "2412 kbps",
                value: "2412"
            }]
            }]
        };

}
