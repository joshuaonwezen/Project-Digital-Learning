    <script>
var myForm,
formData;
function doOnLoad() {
    formData = [{
        type: "settings",
        position: "label-left",
        labelWidth: 90,
        inputWidth: 120
    }, {
        type: "input",
        name: "name",
        label: "Full Name",
        value: "James Belop"
    }, {
        type: "input",
        name: "addr",
        label: "Address",
        value: "22 Acacia Avenue"
    }, {
        type: "checkbox",
        name: "lic",
        label: "I agree with terms and conditions",
        required: true,
        position: "label-right",
        labelWidth: "auto",
        offsetTop: 20
    }, {
        type: "button",
        name: "validate",
        value: "Validate",
        offsetTop: 20,
        offsetLeft: 45
    }];
    myForm = new dhtmlXForm("myForm", formData);
    myForm.attachEvent("onButtonClick", function(name) {
        if (name == "validate") {
            myForm.validate();
        }
    });
}
</script>
